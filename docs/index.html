3分で始められる

HTMLにdata属性を追加するだけ。複雑な設定は不要です。

index.html 

<script src="../dist/FormValidator.min.js"></script>


<form id="contactForm">
    <input type="text" name="name" data-validate="required" placeholder="お名前">
    <input type="email" name="email" data-validate="required,email" placeholder="メール">
    <input type="tel" name="phone" data-validate="tel" placeholder="電話番号">
    
    <p>必須項目: <span data-count_validate></span></p>
    <button type="submit">送信</button>
</form>


<script>
FormValidator.init({
    disableSubmitUntilValid: true
});
</script>
